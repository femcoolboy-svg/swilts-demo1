const express = require('express'); const sqlite3 = require('sqlite3').verbose(); const bcrypt = require('bcrypt'); const path = require('path'); const session = require('express-session'); const SQLiteStore = require('connect-sqlite3')(session); const multer = require('multer'); const http = require('http'); const { Server } = require('socket.io'); const fs = require('fs'); const app = express(); const server = http.createServer(app); const io = new Server(server, { cors: { origin: '*' } }); // ЗАГРУЗКА ФАЙЛОВ const storage = multer.diskStorage({ destination: (req, file, cb) => { const dir = './uploads'; if (!fs.existsSync(dir)) fs.mkdirSync(dir); cb(null, dir); }, filename: (req, file, cb) => { cb(null, Date.now() + '-' + Math.random().toString(36).substr(2) + path.extname(file.originalname)); } }); const upload = multer({ storage, limits: { fileSize: 2 * 1024 * 1024 } }); app.use(express.json()); app.use(express.static('public')); app.use('/uploads', express.static('uploads')); app.use(session({ store: new SQLiteStore({ db: 'sessions.db' }), secret: 'swilts_secret', resave: false, saveUninitialized: false, cookie: { maxAge: 30 * 24 * 60 * 60 * 1000 } })); const db = new sqlite3.Database('swilts.db'); // ТАБЛИЦЫ db.serialize(() => { db.run(`CREATE TABLE IF NOT EXISTS users ( id INTEGER PRIMARY KEY AUTOINCREMENT, username TEXT UNIQUE, password_hash TEXT, tag TEXT UNIQUE, avatar TEXT DEFAULT '', banned INTEGER DEFAULT 0, online INTEGER DEFAULT 0, created_at DATETIME DEFAULT CURRENT_TIMESTAMP )`); db.run(`CREATE TABLE IF NOT EXISTS friends ( id INTEGER PRIMARY KEY AUTOINCREMENT, user_id INTEGER, friend_id INTEGER )`); db.run(`CREATE TABLE IF NOT EXISTS friend_requests ( id INTEGER PRIMARY KEY AUTOINCREMENT, from_id INTEGER, to_id INTEGER )`); db.run(`CREATE TABLE IF NOT EXISTS messages ( id INTEGER PRIMARY KEY AUTOINCREMENT, from_id INTEGER, to_id INTEGER, text TEXT, timestamp DATETIME DEFAULT CURRENT_TIMESTAMP )`); // СОЗДАТЕЛЬ bcrypt.hash('qazzaq32qaz', 10, (err, hash) => { if (!err) { db.run(`INSERT OR IGNORE INTO users (username, password_hash, tag) VALUES (?, ?, ?)`, ['prisanok', hash, '00001']); } }); }); // ============ ОНЛАЙН ПОЛЬЗОВАТЕЛИ ============ const onlineUsers = new Map(); io.on('connection', (socket) => { socket.on('user_online', (userId) => { onlineUsers.set(userId, socket.id); db.run(`UPDATE users SET online = 1 WHERE id = ?`, [userId]); io.emit('user_status', { userId, online: true }); }); socket.on('send_message', (data) => { const messageId = Date.now() + '-' + Math.random(); db.run(`INSERT INTO messages (id, from_id, to_id, text, timestamp) VALUES (?, ?, ?, ?, ?)`, [messageId, data.from, data.to, data.text, new Date().toISOString()]); const targetSocket = onlineUsers.get(data.to); if (targetSocket) { io.to(targetSocket).emit('new_message', { id: messageId, from: data.from, fromName: data.fromName, text: data.text, time: new Date() }); } io.to(socket.id).emit('message_sent', { id: messageId }); }); socket.on('typing', (data) => { const target = onlineUsers.get(data.to); if (target) { io.to(target).emit('user_typing', { from: data.from, name: data.name }); } }); socket.on('typing_stop', (data) => { const target = onlineUsers.get(data.to); if (target) { io.to(target).emit('user_typing_stop', { from: data.from }); } }); socket.on('disconnect', () => { let disconnectedUserId = null; for (let [id, sid] of onlineUsers.entries()) { if (sid === socket.id) { disconnectedUserId = id; onlineUsers.delete(id); break; } } if (disconnectedUserId) { db.run(`UPDATE users SET online = 0 WHERE id = ?`, [disconnectedUserId]); io.emit('user_status', { userId: disconnectedUserId, online: false }); } }); }); // ============ АВТОРИЗАЦИЯ ============ app.post('/register', (req, res) => { const { username, password } = req.body; if (!username || !password) return res.json({ error: 'Заполните поля' }); if (password.length < 6) return res.json({ error: 'Пароль минимум 6 символов' }); db.get(`SELECT id FROM users WHERE username = ?`, [username], (err, row) => { if (row) return res.json({ error: 'Ник занят' }); db.get(`SELECT tag FROM users ORDER BY CAST(tag AS INTEGER) DESC LIMIT 1`, (err, row) => { const nextTag = String((row ? parseInt(row.tag) + 1 : 1)).padStart(5, '0'); bcrypt.hash(password, 10, (err, hash) => { db.run(`INSERT INTO users (username, password_hash, tag) VALUES (?, ?, ?)`, [username, hash, nextTag], function(err) { if (err) return res.json({ error: 'Ошибка' }); req.session.userId = this.lastID; req.session.username = username; res.json({ success: true, userId: this.lastID, username, tag: nextTag }); }); }); }); }); }); app.post('/login', (req, res) => { const { username, password } = req.body; db.get(`SELECT * FROM users WHERE username = ?`, [username], (err, user) => { if (!user) return res.json({ error: 'Неверный логин или пароль' }); if (user.banned) return res.json({ error: `Аккаунт забанен` }); bcrypt.compare(password, user.password_hash, (err, result) => { if (!result) return res.json({ error: 'Неверный логин или пароль' }); req.session.userId = user.id; req.session.username = user.username; res.json({ success: true, userId: user.id, username: user.username, tag: user.tag, avatar: user.avatar }); }); }); }); app.get('/me', (req, res) => { if (!req.session.userId) return res.json({ error: 'Не авторизован' }); db.get(`SELECT id, username, tag, avatar FROM users WHERE id = ?`, [req.session.userId], (err, user) => { res.json(user); }); }); app.post('/logout', (req, res) => { req.session.destroy(); res.json({ success: true }); }); // ============ АВАТАР ============ app.post('/avatar', upload.single('avatar'), (req, res) => { if (!req.file) return res.json({ error: 'Нет файла' }); const url = `/uploads/${req.file.filename}`; db.run(`UPDATE users SET avatar = ? WHERE id = ?`, [url, req.session.userId]); res.json({ success: true, url }); }); // ============ ДРУЗЬЯ ============ app.get('/friends', (req, res) => { if (!req.session.userId) return res.json([]); db.all(`SELECT u.id, u.username, u.tag, u.avatar, u.online FROM friends f JOIN users u ON f.friend_id = u.id WHERE f.user_id = ?`, [req.session.userId], (err, rows) => { res.json(rows || []); }); }); app.get('/requests', (req, res) => { if (!req.session.userId) return res.json([]); db.all(`SELECT fr.id, u.id as user_id, u.username, u.tag FROM friend_requests fr JOIN users u ON fr.from_id = u.id WHERE fr.to_id = ?`, [req.session.userId], (err, rows) => { res.json(rows || []); }); }); app.post('/friend/add', (req, res) => { const { toId } = req.body; if (req.session.userId === toId) return res.json({ error: 'Нельзя добавить себя' }); db.get(`SELECT id FROM friends WHERE (user_id = ?
